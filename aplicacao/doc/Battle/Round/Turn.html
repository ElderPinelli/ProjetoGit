def start

        if dano > 0
                defender.current_hp = defender.current_hp - dano
        end
        
        puts "Nome: #{attacker.name} - HP: #{attacker.current_hp}"
        puts "Nome: #{defender.name} - HP: #{defender.current_hp}"
        puts "-------------------------------------"          
end
